.( ## GAN demo - MNIST ) cr
\ 100 constant N                              \ mini-batch sample count
1 trace
3 constant N                              \ mini-batch sample count

.( ### output file name handlers )
: fn s" ../out/l7_0000" ;                   \ filename storage
fn drop 10 + constant pos                   \ pointer to filename last digits
: n2i ( n i -- )                            \ set n at ith digit
  pos + >r 10 mod $30 + r> c! ;
: setfn ( n -- )                            \ create next filename
  3 for dup i n2i 10 / next drop ;          \ fill 4 digits into fn

.( ### benchmark/statistics ) cr
variable _en 0 _en !                        \ epoch number
variable _bn 0 _bn !                        \ mini-batch counter
variable _gr variable _dr variable _df      \ loss for gen, real, and fake
: stat ( -- ) cr                            \ display loss statistics
  ." t=" clock 1000 / .
  ." , epoch=" _en @ dup >r .
  ." , G="  _gr @ r@ / . 0 _gr !
  ." , Dr=" _dr @ r@ / . 0 _dr !
  ." , Df=" _df @ r> / . 0 _df !
  cr mstat ;
: hint ( -- )                               \ display '.' (dot) for progress 
  _bn @ 1+ dup _bn !
  10 mod 0= if 46 emit then ;
: save_img ( t4 -- )
  _en @ dup 1+ _en ! setfn                  \ create next filename
  fn bin save drop ;                        \ save raw image file

.( ### create discriminator network ) cr
N 28 28 1 nn.model                          \ define Discriminator model (8-layer)
512 linear 0.2 leakyrelu 0.3 dropout        \ 1st linear layer
256 linear 0.2 leakyrelu 0.3 dropout        \ 2nd linear layer
1 linear sigmoid                            \ binary output layer
constant D                                  \ kept as constant D

.( ### create generator network ) cr
N 128 1 1 nn.model                          \ define Generator Model (6-layer)
256 linear 0.2 leakyrelu                    \ 1st linear layer
512 linear 0.2 leakyrelu                    \ 2nd linear layer
784 linear tanh                             \ 28x28x1, (1+tanh)/2 output [0,1)
constant G                                  \ kept as constant G

.( ### our entire GAN here )
0.00001 constant d_lr 0.00001 constant g_lr \ init learning rates
N dataset mnist_train                       \ create dataset MNIST (Real data)
\ constant ds0                                \ kept as a constant

bye

: REAL N 1 1 1 tensor ones ;                \ create a real label tensor
: FAKE N 1 1 1 tensor zeros ;               \ create a fake label tensor
: seed N 128 1 1 tensor randn ;             \ create randn avg=0,std=1 input for fake data
: gen ( -- t4 )                             \ generate a mini-batch of fake images
  G seed forward -1 n@                      \ feed random seeds to generator
  N 28 28 1 reshape4                        \ reshape to [N, 28, 28, 1]
  swap drop ;                               \ drop view of G

: train_d ( D -- )
  1 trainable                                     \ ensure D trainable
  ds0 forward REAL loss.bce _dr +! REAL backprop  \ feed D with real data (from rs)
  gen forward FAKE loss.bce _df +! FAKE backprop  \ feed D with fake data, update D
  d_lr 0.5 nn.adam ;                              \ train D, Adam (b1=0.5,b2=1-lr)
: train_g ( D -- )
  0 trainable                                     \ D testing only
  gen forward REAL loss.bce _gr +! REAL backprop  \ feed D as real (but without updating)
  1 n@ G swap ( D G t ) backprop                  \ propagate image dX to G backprop
  g_lr 0.5 nn.adam ( D G ) drop ;                 \ train G, Adam (b1=0.5,b2=1-lr)
  
: epoch ( D ds -- D )                       \ one epoch for the given data source
  for                                       \ loop thru mini-batches (on rs)
    train_d train_g hint                    \ training and display '.' for progress
  next ( D ) 0 _bn ! ;                      \ loop, if next mini-batch available
: gan ( D ds n -- D ds )
  for
    epoch stat gen save_img                 \ create image dump per epoch
    d_lr 0.95 * [to] d_lr                   \ D learning rate decay 
    g_lr 0.95 * [to] g_lr                   \ D learning rate decay 
    ds0 rewind ( D ds )                     \ rewind dataset 
  next ;

\ expected loss G=>low, Dr=>low, Df=>high (thinking fake as real)
D ds0 epoch \ 99 gan

bye
