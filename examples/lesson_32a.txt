.( ## GAN tests Z = A X + B ) cr        \ linear regression in matrix form
100 constant N                          \ mini-batch size (100 points)

.( ### build generator network )
N 2 1 1 nn.model                        \ generator model
2 linear                                \ just one linear layer, w=2x2 (bias=1.0)
constant G                              \ kept as a constant

.( ### build discriminator network )
N 2 1 1 nn.model                        \ discriminator model
5 linear 0.2 leakyrelu                  \ 1st linear layer
3 linear 0.2 leakyrelu                  \ 2nd linear layer
1 linear sigmoid                        \ binary output layer
constant D                              \ kept as a constant

.( ### regression math )
2 2 matrix{ 1 2 -1 -2 } constant A      \ linear transformation matrix
N 2 matrix ones                         \ N data points
2 2 matrix{ 1 0 0 2 } @= constant B     \ create offset matrix
: X ( -- t4 ) N 2 matrix randn ;        \ random input N points of { x1, x2 }
: Z ( -- t4 ) X A @= B += ;             \ create real samples i.e. Z = AX + B

N 1 1 1 tensor ones constant  REAL      \ onehot tersor for a real set
N 1 1 1 tensor zeros constant FAKE      \ onehot tensor for a fake set

: gen ( -- t4 )                         \ run generator (faking one set of samples)
  G X forward -1 n@ swap drop ;

0 constant _g 0 constant _r 0 constant _f         \ loss for gen, real, and fake
\ expected loss G=>low, Dr=>low, Df=>high (thinking fake as real)
: stat ( -- )                                     \ display statistics
  cr ." w,b=" G 1 nn.w . 1 nn.b . drop
  ." G=" _g . ." , Dr=" _r . ." , Df=" _f . cr ;

: run ( -- )
  D 1 trainable                                   \ make D discriminator trainable
  Z   forward REAL loss.bce [to] _r REAL backprop \ treat real samples as real
  gen forward FAKE loss.bce [to] _f FAKE backprop \ treat fake samples as fake
  0.001 nn.adam ( D )                             \ train D, Adam (b1=0.9,b2=0.999)
  0 trainable                                     \ turn D back to testing mode
  gen forward REAL loss.bce [to] _g REAL backprop \ now treat fake samples as real
  1 n@ ( D t ) G swap backprop                    \ propagate dX back to G
  0.001 nn.adam ( D G )                           \ refine/train G with Adam
  drop drop                                       \ drop views of D, G models
  ;

: epoch 99 for run next ;                         \ train 100 * N samples
: gan ( n -- ) 1-                                 \ run n epoch
  for epoch stat next ;                                             

0 trace                                           \ debug verbosity
10 gan                                            \ run multiple (10) epochs
gen N 2 reshape2 s" l32a.out" save                \ generate N trained { x1, x2 }

bye
